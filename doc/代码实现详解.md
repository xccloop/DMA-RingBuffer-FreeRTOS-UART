## 2. RingBuffer 模块详解

### 2.1 数据结构 (`ringbuffer.h:19-24`)

```c
typedef struct {
    volatile uint32_t head;     // 写指针（生产者独占修改）
    volatile uint32_t tail;     // 读指针（消费者独占修改）
    uint8_t         *buffer;    // 外部提供的存储区
    uint32_t         mask;      // = capacity - 1（必须 capacity 为 2 的幂）
    uint32_t         capacity;  // 缓冲区总容量
} ringbuffer_t;
```

**设计要点**：
- `head` 和 `tail` 都是单调递增的 32 位计数器，**不是** 0~capacity-1 的循环索引
- 通过 `head & mask` 得到数组索引（位掩码取模）
- `volatile` 确保每次访问都从内存读取，防止 ISR→Task 可见性问题

### 2.2 初始化 (`ringbuffer.c:7-16`)

```c
void rb_init(ringbuffer_t *rb, uint8_t *buf, uint32_t capacity)
{
    if (!rb || !buf || !is_power_of_two(capacity)) {
        return;  // 静默拒绝非法参数
    }
    rb->buffer   = buf;
    rb->capacity = capacity;
    rb->mask     = capacity - 1;
    rb->head     = 0;
    rb->tail     = 0;
}
```

幂等检查 `is_power_of_two`：
```c
static bool is_power_of_two(uint32_t x)
{
    return (x > 0) && ((x & (x - 1)) == 0);
}
```
例如：`256 & 255 = 0` → 是 2 的幂；`500 & 499 = 496 ≠ 0` → 不是。

### 2.3 写入操作 (`ringbuffer.c:18-44`)

```c
uint32_t rb_put(ringbuffer_t *rb, const uint8_t *data, uint32_t len)
{
    // 1. 计算可用空间，截断超出的数据
    uint32_t free = rb_free_space(rb);   // capacity - (head - tail)
    if (len > free) len = free;
    
    // 2. 当前写入位置索引
    uint32_t idx = head & rb->mask;
    
    // 3. 处理回绕：数据可能分两段写入
    uint32_t seg1 = rb->capacity - idx;  // 从 idx 到缓冲区末尾
    if (seg1 >= len) {
        memcpy(&rb->buffer[idx], data, len);        // 单段
    } else {
        memcpy(&rb->buffer[idx], data, seg1);       // 第一段
        memcpy(&rb->buffer[0], data + seg1, len - seg1); // 第二段 (回绕)
    }
    
    // 4. 更新 head（只用 head += len，不做取模）
    rb->head = head + len;
    return len;
}
```

### 2.4 连续长度查询 (DMA 分段传输的关键)

```c
uint32_t rb_contig_read_len(const ringbuffer_t *rb)
{
    uint32_t avail = rb_available(rb);    // head - tail
    uint32_t idx   = rb->tail & rb->mask;
    uint32_t seg1  = rb->capacity - idx;  // 从 tail 到缓冲区末尾的长度
    return (avail < seg1) ? avail : seg1;
}
```

DMA TX 调用此函数确定一次 DMA 传输可以覆盖多少连续字节：

```
场景 A (无回绕)：tail=2, head=7, capacity=8
  idx=2, seg1=6, avail=5
  → contig_read_len = min(5, 6) = 5
  → DMA 一次传输 5 字节: [2, 7)

场景 B (有回绕)：tail=5, head=10, capacity=8
  idx=5, seg1=3, avail=5
  → contig_read_len = min(5, 3) = 3
  → DMA 第一段传输 3 字节: [5, 8)
  → DMA 第二段传输 2 字节: [0, 2)
```

## 3. Port 层详解 (`port/stm32f1xx/uart_port.c`)

### 3.1 端口控制块

```c
typedef struct {
    USART_TypeDef            *usart;
    DMA_Channel_TypeDef      *dma_tx_ch;      // TX DMA 通道寄存器
    DMA_Channel_TypeDef      *dma_rx_ch;      // RX DMA 通道寄存器
    IRQn_Type                 usart_irq;
    IRQn_Type                 dma_tx_irq;
    IRQn_Type                 dma_rx_irq;

    // 由驱动层注册的 ISR 回调
    uart_port_tx_tc_cb_t      tx_tc_cb;        // TX 传输完成
    uart_port_tx_ht_cb_t      tx_ht_cb;        // TX 半传输
    uart_port_rx_tc_cb_t      rx_tc_cb;        // RX 传输完成
    uart_port_rx_ht_cb_t      rx_ht_cb;        // RX 半传输
    uart_port_idle_cb_t       idle_cb;         // UART IDLE
    uart_port_error_cb_t      error_cb;        // 错误

    volatile bool             tx_dma_busy;
    volatile bool             rx_dma_busy;

    uint32_t                  rx_circ_len;       // 记录循环模式的缓冲区长度
} uart_port_t;

static uart_port_t g_ports[UART_PORT_MAX];  // 全局静态数组
```

### 3.2 初始化流程 (`uart_port_init`)

```c
void uart_port_init(uint8_t port, uint32_t baudrate)
{
    // step 1: 清零控制块
    memset(&g_ports[port], 0, sizeof(uart_port_t));
    
    // step 2: 获取端口对应的硬件地址
    uart_port_get_hal_info(port, &usart, &tx_ch, &rx_ch, ...);
    
    // step 3: 初始化 GPIO + 时钟
    uart_port_msp_init(port);
    
    // step 4: 计算并设置波特率
    uint32_t pclk = (port == 0) ? HAL_RCC_GetPCLK2Freq() : HAL_RCC_GetPCLK1Freq();
    usart->BRR = (pclk + baudrate / 2) / baudrate;
    
    // step 5: 配置 USART：TX+RX 使能，DMA 触发，IDLE 中断
    usart->CR1 = USART_CR1_TE | USART_CR1_RE | USART_CR1_IDLEIE;
    usart->CR3 = USART_CR3_DMAT | USART_CR3_DMAR;
    
    // step 6: 配置 DMA 通道
    // TX: mem→periph, 8-bit, normal mode, memory increment, TC interrupt
    tx_ch->CCR = DMA_CCR_MINC | DMA_CCR_DIR | DMA_CCR_PL | DMA_CCR_TCIE;
    // RX: periph→mem, 8-bit, circular mode, TC + HT interrupt
    rx_ch->CCR = DMA_CCR_MINC | DMA_CCR_CIRC | DMA_CCR_PL_1 | DMA_CCR_TCIE | DMA_CCR_HTIE;
    
    // step 7: NVIC 优先级（DMA < USART，确保 IDLE 不被 DMA ISR 过度打断）
    NVIC_SetPriority(usart_irq, 6);
    NVIC_SetPriority(tx_irq,    7);
    NVIC_SetPriority(rx_irq,    7);
}
```

### 3.3 DMA 启动 (`uart_port_dma_tx_start`)

```c
void uart_port_dma_tx_start(uint8_t port, const uint8_t *data, uint32_t len)
{
    // step 1: 先停 DMA（必须，因为 EN=1 时不能改 CMAR/CNDTR）
    p->dma_tx_ch->CCR &= ~DMA_CCR_EN;
    
    // step 2: 配置外设地址 (USART DR) 和内存地址
    p->dma_tx_ch->CPAR  = (uint32_t)&p->usart->DR;
    p->dma_tx_ch->CMAR  = (uint32_t)data;
    p->dma_tx_ch->CNDTR = (uint16_t)len;
    
    // step 3: 使能 DMA 中断，启动传输
    NVIC_EnableIRQ(p->dma_tx_irq);
    p->dma_tx_ch->CCR |= DMA_CCR_EN;
    p->tx_dma_busy = true;
}
```

### 3.4 中断处理 (`uart_port_irq_handler`)

```c
void uart_port_irq_handler(uint8_t port)
{
    uint32_t sr = p->usart->SR;
    
    // IDLE 中断
    if ((sr & USART_SR_IDLE) && (p->usart->CR1 & USART_CR1_IDLEIE)) {
        (void)p->usart->DR; // 读 DR 清除 IDLE
        if (p->idle_cb) p->idle_cb(port);
    }
    
    // 错误处理：PE(奇偶错)、FE(帧错)、NE(噪声)、ORE(溢出)
    if (sr & (USART_SR_PE | USART_SR_FE | USART_SR_NE | USART_SR_ORE)) {
        (void)p->usart->DR; // 读 DR 清除错误标志
        if (p->error_cb) p->error_cb(port, sr & 0x0F);
    }
}
```

### 3.5 DMA 中断处理 (`uart_port_dma_tx_irq_handler`)

```c
void uart_port_dma_tx_irq_handler(uint8_t port)
{
    // DMA1 中断标志寄存器是全局的，需要根据通道号判断
    uint32_t tc_mask = DMA_ISR_TCIF1 << ((ch_idx - 1) * 4);
    
    if (DMA1->ISR & tc_mask) {
        DMA1->IFCR = tc_mask;  // 写 1 清除标志
        p->tx_dma_busy = false;
        NVIC_DisableIRQ(p->dma_tx_irq);
        
        if (p->tx_tc_cb) p->tx_tc_cb(port);
    }
}
```

**DMA 通道号 → ISR 位偏移对应表**：

| 通道 | TC 标志位 | 偏移量 | 清除位 |
|------|-----------|--------|--------|
| CH1 | ISR[1] | (1-1)*4=0 | IFCR[0] |
| CH2 | ISR[5] | (2-1)*4=4 | IFCR[4] |
| CH3 | ISR[9] | (3-1)*4=8 | IFCR[8] |
| CH4 | ISR[13] | (4-1)*4=12 | IFCR[12] |
| CH5 | ISR[17] | (5-1)*4=16 | IFCR[16] |
| CH6 | ISR[21] | (6-1)*4=20 | IFCR[20] |
| CH7 | ISR[25] | (7-1)*4=24 | IFCR[24] |

### 4.1 驱动句柄

```c
struct uart_dma_handle {

    ringbuffer_t  rx_rb;              // RX 环形缓冲区
    TaskHandle_t  tx_task;            // 等待 TX 完成的任务
    TaskHandle_t  rx_task;            // 等待 RX 数据的任务
    volatile uint32_t tx_kick_head;   // TX kick 时的 head 快照
    volatile bool     tx_chain_pending;  // 是否需要第二段 DMA
    volatile uint32_t rx_overrun_cnt;    // RX 溢出计数器
    volatile uint32_t error_flags;       // 累积的错误标志
    volatile bool     tx_idle;           // TX DMA 是否空闲
};
```

### 4.2 TX 链路 — 核心算法 (`dma_tx_kick`)

此函数是 TX 路径的心跳。每当地址可用的发生改变（新数据到达 + DMA 空闲），它就会被调用：

```c
static void dma_tx_kick(uart_dma_handle_t *h)
{
    uint32_t head = rb_get_head(&h->tx_rb);
    uint32_t tail = rb_get_tail(&h->tx_rb);

    if (head == tail) {
        h->tx_idle = true;    // 无数据，DMA 进入空闲
        return;
    }

    h->tx_kick_head = head;   // ★ 快照 head，后续 TC 中断以此为界
    h->tx_idle = false;

    uint32_t tail_idx = tail & h->tx_rb.mask;
    uint32_t head_idx = head & h->tx_rb.mask;

    if (tail < head) {
        // ====== 情况1: 无回绕，单段传输 ======
        // 数据在 [tail, head) 之间是连续的
        uint32_t seg_len = head - tail;
        h->tx_chain_pending = false;
        uart_port_dma_tx_start(h->port, &h->tx_rb.buffer[tail_idx], seg_len);
        //                              ↑ 直接传 buffer 中的地址
    } else {
        // ====== 情况2: 有回绕，分两段传输 ======
        // 数据分裂为 [tail, capacity) + [0, head)
        uint32_t seg1_len = h->tx_rb.capacity - tail_idx;
        h->tx_chain_pending = (head_idx > 0);
        uart_port_dma_tx_start(h->port, &h->tx_rb.buffer[tail_idx], seg1_len);
    }
}
```

**不回绕 vs 回绕图解**：

```
容量 = 8, mask = 7

不回绕: head=7, tail=2  (head > tail)
  0     1     2     3     4     5     6     7
  ┌─────┬─────┬─────┬─────┬─────┬─────┬─────┬─────┐
  │     │     │ DAT │ DAT │ DAT │ DAT │ DAT │     │
  └─────┴─────┴─────┴─────┴─────┴─────┴─────┴─────┘
              ▲ tail=2                    ▲ head=7
  连续段: [2, 7) = 5 字节 → 一次 DMA

回绕: head=10, tail=5  (head > tail 但索引回绕了)
  0     1     2     3     4     5     6     7
  ┌─────┬─────┬─────┬─────┬─────┬─────┬─────┬─────┐
  │ DAT │ DAT │     │     │     │ DAT │ DAT │ DAT │
  └─────┴─────┴─────┴─────┴─────┴─────┴─────┴─────┘
  ▲ head_idx=2           ▲ tail=5
  第一段: [5, 8) = 3 字节 → DMA1
  第二段: [0, 2) = 2 字节 → DMA2 (TC 中断启动)
```

### 4.3 TX TC 中断处理 (`dma_tx_tc_isr`)

```c
static void dma_tx_tc_isr(uint8_t port)
{
    if (h->tx_chain_pending) {
        // ***** 第二段传输 *****
        // 前进 tail 过第一段
        h->tx_rb.tail = (h->tx_rb.tail & ~h->tx_rb.mask) + h->tx_rb.capacity;
        // 钳位: 不超过快照 head
        if (h->tx_rb.tail > h->tx_kick_head) h->tx_rb.tail = h->tx_kick_head;

        // 启动第二段 [0, kick_head)
        uint32_t seg2_len = h->tx_kick_head & h->tx_rb.mask;
        if (seg2_len > 0) {
            uart_port_dma_tx_start(h->port, &h->tx_rb.buffer[0], seg2_len);
            h->tx_chain_pending = false;  // 第二段后不会再有链
        }
    } else {
        // ***** 单段 / 第二段 完成 *****
        h->tx_rb.tail = h->tx_kick_head;

        // 检查在传输期间是否有新数据到达
        uint32_t cur_head = rb_get_head(&h->tx_rb);
        uint32_t cur_tail = rb_get_tail(&h->tx_rb);

        if (cur_head != cur_tail) {
            dma_tx_kick(h);  // 有新数据 → 继续 kick
        } else {
            h->tx_idle = true;
            // 通知等待的 TX 任务
            vTaskNotifyGiveFromISR(h->tx_task, &xHigherPriorityTaskWoken);
        }
    }
}
```

### 4.4 RX IDLE 中断处理 (`dma_idle_isr`)

```c
static void dma_idle_isr(uint8_t port)
{
    // IDLE 标志已由 port 层清除，这里只需读 NDTR
    uint32_t ndtr     = uart_port_rx_dma_remaining(port);
    uint32_t capacity = h->rx_rb.capacity;
    uint32_t received = capacity - ndtr;  // DMA 已传输的字节总数

    // 将 DMA 的绝对位置映射到 ringbuffer 的单调 head
    uint32_t old_head = rb_get_head(&h->rx_rb);
    uint32_t modulo   = old_head % capacity;
    uint32_t base     = old_head - modulo;

    uint32_t new_head;
    if (received > modulo) {
        new_head = base + received;            // 正常前进
    } else if (received < modulo) {
        new_head = base + capacity + received; // DMA 回绕了一圈
    } else {
        return;  // received == modulo: 没有新数据
    }

    // 推进 head 并通知任务
    uint32_t advance = new_head - old_head;
    rb_advance_head(&h->rx_rb, advance);
    
    vTaskNotifyGiveFromISR(h->rx_task, &xHigherPriorityTaskWoken);
}
```

**NDTR → head 的位置映射**是这个函数最微妙的逻辑：

DMA 循环模式的 CNDTR 从 `capacity` 递减到 `0` → 自动回绕到 `capacity`。所以从 CNDTR 得到的 `received = capacity - NDTR` 是 DMA 在当前周期中的绝对偏移。

而 RingBuffer 的 `head` 是单调递增的（从不回绕），所以需要把 DMA 的循环偏移映射到单调 head：

```
例: capacity = 1024, 当前 DMA 已传输 300 字节

第一次 IDLE: old_head=0, received=300
  modulo=0, base=0
  received(300) > modulo(0) → new_head = 0 + 300 = 300
  推进: head 从 0 到 300 ✓

一段时间后，DMA 不断循环...
DMA 回绕了 2 圈后传了 500 字节

第二次 IDLE: old_head=300, received=500
  modulo=300, base=0
  received(500) > modulo(300) → new_head = 0 + 500 = 500
  但是... 正确的 head 应该是 2*1024 + 500 = 2548!

  这里确实有个问题: 如果 Task 消费速度 < DMA 接收速度,
  head 可能会落后 DMA 若干圈。解决方案是: HT/TC 中断
  会在每半满/全满时更新 head，确保 head 始终与 DMA 同步。
  IDLE 只负责最后几个不完整的半满周期。
```

### 5.2 初始化 (`uart_init`)

```c
void uart_init(uint8_t port, uint32_t baudrate,
               uint32_t tx_buf_size, uint32_t rx_buf_size)
{
    // 1. 动态分配 RingBuffer 存储区
    ctx->tx_buf_storage = pvPortMalloc(tx_buf_size);
    ctx->rx_buf_storage = pvPortMalloc(rx_buf_size);
    
    // 2. 创建 Mutex
    ctx->tx_mutex = xSemaphoreCreateMutex();
    
    // 3. 打开驱动（内部初始化 port 层 + 启动 RX DMA）
    ctx->drv = uart_dma_open(port, baudrate,
        ctx->tx_buf_storage, tx_buf_size,
        ctx->rx_buf_storage, rx_buf_size);
}
```

### 5.3 阻塞发送 (`uart_send`)

```c
int32_t uart_send(uint8_t port, const uint8_t *data, uint32_t len, uint32_t timeout)
{
    xSemaphoreTake(ctx->tx_mutex, timeout);  // 获取互斥锁
    
    while (total_sent < len) {
        free_space = uart_dma_tx_free_space(ctx->drv);
        
        if (free_space == 0) {
            // RingBuffer 满 → 短暂释放锁，yield，重试
            xSemaphoreGive(ctx->tx_mutex);
            vTaskDelay(pdMS_TO_TICKS(UART_TX_RETRY_MS));
            xSemaphoreTake(ctx->tx_mutex, timeout);
            continue;
        }
        
        // 写入尽可能多的数据
        chunk = min(remaining, free_space);
        written = uart_dma_send(ctx->drv, data + total_sent, chunk);
        total_sent += written;
    }
    
    xSemaphoreGive(ctx->tx_mutex);  // 释放互斥锁
    return total_sent;
}
```

### 5.4 阻塞接收 (`uart_recv`)

```c
int32_t uart_recv(uint8_t port, uint8_t *buf, uint32_t len, uint32_t timeout)
{
    while (1) {
        avail = uart_dma_rx_available(ctx->drv);
        if (avail > 0) {
            return uart_dma_recv(ctx->drv, buf, min(len, avail));
        }
        
        // 没有数据 → 等待 ISR 通知
        ulTaskNotifyTake(pdTRUE, wait_ticks);
        // 被 IDLE/HT/TC ISR 唤醒后，继续循环检查
    }
}
```

### 5.5 格式化输出 (`uart_printf`)

```c
int32_t uart_printf(uint8_t port, const char *fmt, ...)
{
    static char printf_buf[UART_PRINTF_BUF_SIZE];  // 编译期分配，避免栈溢出
    
    va_list args;
    va_start(args, fmt);
    int len = vsnprintf(printf_buf, sizeof(printf_buf), fmt, args);
    va_end(args);
    
    return uart_send(port, (const uint8_t *)printf_buf, (uint32_t)len, portMAX_DELAY);
}
```

### 5.6 帧回调机制 (`uart_yield`)

```c
void uart_yield(uint8_t port, uint32_t timeout)
{
    while (1) {
        avail = uart_dma_rx_available(ctx->drv);
        
        if (avail > 0 && ctx->rx_cb) {
            // IDLE 之后的所有数据作为一个"帧"递交给回调
            uint8_t frame_buf[256];
            uint32_t n = uart_dma_recv(ctx->drv, frame_buf, min(avail, 256));
            ctx->rx_cb(port, frame_buf, n, ctx->rx_cb_arg);
        }
        
        // 等待下一次 IDLE
        if (ulTaskNotifyTake(pdTRUE, wait_ticks) == 0) {
            // 超时，检查是否要退出循环
            if (timeout expired) break;
        }
    }
}
```
